原题：[C. Dora and C++](https://codeforces.com/contest/2007/problem/C)
## 题意

给一个数组 $c$ ，两个数字 $a$ 和 $b$，任选 $c_i$ 加 $a$ 或 $b$，最小化 $max(c)-min(c)$

## 思路

由[[扩展欧几里得]]，也就是 $ax+by=c$ 得两个数的差值最小是 $c$，也就是 $gcd(a,b)$

$c$ 的所有元素在若干次操作后都相当于加了 $gcd$

所以差值都在 [[../../../数据结构与算法/数学/GCD]] 内，将 $c_i=c_i \% gcd$ 

然后计算[[极差]]

## 代码

```CPP
##define rep(i, j, k) for (int i = (j); i < (k); i++)
##define vi vector<int>

vi arr;
int solve() {
    int n, a, b;
    int mx = 0;
    int mi = 1e9;
    cin >> n >> a >> b;
    arr.resize(n);
    rep(i, 0, n) cin >> arr[i];

    if (mx == mi) return 0;
    int len = abs(a - b);
    if (len != 0)
        len = __gcd(a, b);
    else
        len = a;

    for (int &i : arr) i %= len;

    sort(all(arr));
    int ans = arr[n - 1] - arr[0];
    for (int i = 0; i < n - 1; i++) ans = min(ans, arr[i] + len - arr[i + 1]);

    return ans;
}

signed main() {
    IOS;
    int t = 1;
    cin >> t;
    while (t--) cout << solve() << endl;
    return 0;
}
```
