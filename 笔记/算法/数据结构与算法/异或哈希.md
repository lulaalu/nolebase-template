一种，及其好理解的算法

就是，由于相同的数异或等于0，可以用来统计奇数的存在与否

但是由于1^2^3=0，可是这都是奇数，那用一个极大的随机数取代原本的数就行


例题：[Codeforces Round 974 (Div. 3) H. Robin Hood Archery](https://codeforces.com/contest/2014)

题意：求区间内是否有出现奇数次的数

# CODE

```CPP
// H. Robin Hood Archery
// Codeforces Round 974 (Div. 3) https://codeforces.com/contest/2014
#pragma GCC optimize(2)
#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define IOS ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#define rep(i, j, k) for (int i = (j); i < (k); i++)
#define all(x) x.begin(), x.end()
#define pii pair<int, int>
#define ull unsigned long long

unsigned seed1 = std::chrono::system_clock::now().time_since_epoch().count();
std::mt19937_64 g(seed1);

const int N = 2e5 + 10;
ull hsum[N];
ull ha[1000001];

void solve() {
    int n, m;
    cin >> n >> m;
    hsum[0] = 0;
    rep(i, 1, n + 1) {
        int x;
        cin >> x;
        hsum[i] = ha[x];
        hsum[i] ^= hsum[i - 1];
    }
    rep(i, 0, m) {
        int l, r;
        cin >> l >> r;
        // cout << "l,r:" << l << "," << r << " XOR:" << (hsum[l - 1] ^ hsum[r]) << endl;
        if ((hsum[l - 1] ^ hsum[r]) == (ull)0)
            cout << "YES" << endl;
        else
            cout << "NO" << endl;
    }
}
int main() {
    IOS;
    srand(time(0));
    for (int i = 0; i <= 1e6; i++) ha[i] = g();
    int t;
    cin >> t;
    while (t--) solve();
    return 0;
}
```
