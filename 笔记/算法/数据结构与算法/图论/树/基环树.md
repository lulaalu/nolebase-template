# 基本概念

$N$ 个点的树有 $N-1$ 条边，如果再加一条边，会出现环

这样的**联通无向图**称为**基环树**

如果是不联通的图，那么称为**基环树森林**

当然，如果是有向图，并且每个点入度为1，也叫基环树

不过有**外向树**和**内向树**两种

# 基本解题方式

基环树通常作为经典模型的扩展，以增加题目难度

比如基环树的直径、基环树两点间的距离，基环树动态规划

但是一般都是先找出环，以环为基环树的广义“根节点”，环之外的部分按照树处理

# [358. 岛屿](https://www.acwing.com/problem/content/360/)

## 题意

N个点，无向图

给你若干座桥

基本要求：
1. 只能去没去过的岛
2. 步行走桥
3. 渡船：你可以选择这种方法，仅当没有任何桥和以前使用过的渡船的组合可以由 $S$ 走到 $D$（当检查是否可到达时，你应该考虑所有的路径，包括经过你曾游览过的那些岛）。

第三点就是说，整个岛屿群大概是个无向图森林，假设你已经坐船从一颗树$T_{1}$到了另一颗树$T_{2}$

有 $T_{1}$ 内的某一个岛 $D_{x}$ 是你没走过的，理论上你能走到，只是为了最长路径没去走

也就是说有一个组合你使用过的渡船和桥的组合能到达 $D_{x}$ ，只是你没去，那么你就不能去了

就是**求每个基环树的直径/最长链和**
## 思路

基环树的最长链有两种可能
1. 去除环后的某个子树内
2. 经过环

流程：
- [[../../DFS]]找出环，记作$[s_{1},s_{2},\cdots,s_{x}]$
- 从环上的结点出发，用求[[树的直径]]中的[[树形dp]]的方式处理出$ans$和 $d$ 数组
- 根据 $d$ 数组和$dist(s_{i},s_{j})$ 求出$ans=max(ans,d[s_{i}]+d[s_{j}]+dist(s_{i},s_{j}))$ 
	- $dist$ 可以通过断开链然后两倍链获取

#include <bits/stdc++.h>

using i64 = long long;
using i32 = int;

const i32 N = 2e6 + 20;
i32 n, a, l;

set<i32> g[N];
i32 Head[N], To[N], Next[N];
i64 Weight[N];
i32 cnt = 1;
void addEdge(i32 u, i32 v, i64 w)
{
    Next[cnt] = Head[u];
    To[cnt] = v;
    Weight[cnt] = w;
    Head[u] = cnt++;
}

i32 fa[N];
i32 vis_fa[N]; // 表示对应的点遍历过没
i32 Find(int x)
{
    if (x == fa[x])
        return x;
    else
        return fa[x] = Find(fa[x]);
}
void Union(i32 x, i32 y)
{
    i32 fx = Find(x);
    i32 fy = Find(y);
    if (fx != fy)
    {
        fa[fx] = Find(fy);
    }
}

i32 vis1[N]; // 访问次数
i32 s[N], s_cnt = 0;
bool dfs1(int x, int f)
{
    if (vis1[x] == 1)
    {
        s[++s_cnt] = x;
        return true;
    }
    vis1[x] = 1;
    // printf("dfs1(%d, %d, %d) vis1[x]=%d\n", x, f, ff, vis1[x]);

    for (int e = Head[x]; e && !s_cnt; e = Next[e])
    {
        int y = To[e];
        if (y == f)
            continue;
        // cout << "e=" << e << " y=" << y << endl;
        if (dfs1(y, x))
        {
            if (s_cnt > 0 && x == s[1])
                return false;
            else
            {
                // cout << "x=" << x << endl;
                vis1[x] = 2;
                s[++s_cnt] = x;
                return true;
            }
        }
    }
    return false;
}

// 用于处理出d数组以及ans1
bool vis2[N];
i64 d[N];
// 当前点、父、初始环上的点
void dp(i32 x, i32 f, i64 &ans)
{
    // printf("dfs2(%d,%d,%d,%lld)\n", x, f, ss, ans);
    vis2[x] = 1;
    for (int e = Head[x]; e; e = Next[e])
    {
        int y = To[e];
        if (y == f || vis2[y])
            continue;
        dp(y, x, ans);
        ans = max(ans, d[x] + d[y] + Weight[e]);
        d[x] = max(d[x], d[y] + Weight[e]);
    }
}

void init(int n)
{
    fill_n(Head, 2 * n + 10, 0);
    fill_n(To, 2 * n + 10, 0);
    fill_n(Next, 2 * n + 10, 0);
    fill_n(Weight, 2 * n + 10, 0);
    fill_n(vis1, n + 10, 0);
    fill_n(vis2, n + 10, 0);
    fill_n(vis_fa, n + 10, 0);
    fill_n(d, n + 10, 0);
    for (i32 i = 0; i <= n + 10; i++)
    {
        fa[i] = i;
    }
}
i32 q[N];
i64 sum[N];
i64 getw(i32 x, i32 y)
{
    i64 ans = 0;
    for (int e = Head[x]; e; e = Next[e])
    {
        if (To[e] == y)
            ans = max(ans, Weight[e]);
    }
    return ans;
}
signed main()
{
    cin >> n;
    init(n);
    //! 建图并且并查集
    for (i32 i = 1; i <= n; i++)
    {
        cin >> a >> l;
        addEdge(i, a, l);
        addEdge(a, i, l);
        Union(i, a);
    }

    //! 依照并查集，一颗一颗的处理所有基环树
    i64 all_ans = 0;
    for (i32 i = 1; i <= n; i++)
    {
        if (vis_fa[Find(i)])
            continue;
        vis_fa[Find(i)]++;

        //! 第一步：dfs1找出所有的s，[s1,s2,...,s_{s_cnt}]
        s_cnt = 0;
        dfs1(i, 0);

        //! 第二步：dfs2遍历环上所有的点，找出去除环后所有树中的最长链ans1
        for (i32 i = 1; i <= s_cnt; i++)
            vis2[s[i]] = 1;
        i64 ans1 = 0;
        for (i32 i = 1; i <= s_cnt; i++)
        {
            dp(s[i], 0, ans1);
            // cout << s[i] << ' ';
        }
        // cout << endl;
        //! 第三步：处理经过环的最长链
        //* 利用优先队列处理子数组最小值的方式，先断开链再延长一倍
        for (i32 i = 1; i <= s_cnt; i++)
        {
            s[i + s_cnt] = s[i];
        }
        sum[0] = 0;
        sum[1] = 0;
        for (i32 i = 2; i <= 2 * s_cnt; i++)
        {
            sum[i] = getw(s[i], s[i - 1]);
            sum[i] += sum[i - 1];
        }

        i32 hh = 0, tt = -1;
        i64 ans2 = 0;
        for (i32 i = 1; i <= 2 * s_cnt; i++)
        {
            if (hh <= tt && i - q[hh] >= s_cnt)
                hh++;
            // cout << q[hh] << ' ' << i << endl;
            if (hh <= tt && ans2 < d[s[i]] + d[s[q[hh]]] + sum[i] - sum[q[hh]])
            {
                ans2 = d[s[i]] + d[s[q[hh]]] + sum[i] - sum[q[hh]];
                // cout << "ans2对应两点：" << q[hh] << ' ' << i << endl;
            }
            while (hh <= tt && d[s[q[tt]]] - sum[q[tt]] <= d[s[i]] - sum[i])
                tt--;
            q[++tt] = i;
        }
        all_ans += max(ans1, ans2);
    }
    cout << all_ans << endl;
    return 0;
}
