# 概念

- 重儿子：父结点的所有儿子中子树结点数目最多的结点
- 轻儿子：父结点中除重儿子以外的儿子
- 重边：父结点和重儿子连成的边
- 轻边：父结点和轻儿子连成的边
- 重链：由多条重边连接而成的路径

# 性质

1. 整棵树会被剖分成若干条重链.
2. 轻儿子一定是每条重链的顶点
3. 任意一条路径被切分成不超过logn条链

# 原理

大致原理就是类似倍增，当较深的结点往上跳时（指的是当前节点所在的重链的顶端结点之间的比较），最终会跳到同一条重链上，那么较深的结点就是初始两点的lca

# 代码

结构：

```cpp
vector<i32> g[N];
i32 fa[N];    // 存u的父节点
i32 depth[N]; // 存u的深度
i32 sz[N];    // 存以u为根的子树的结点数

i32 son[N]; // 存u的重儿子
i32 top[N]; // 存u所在的重链的顶点

```

```cpp

// 定义一个深度优先搜索函数dfs1，用于遍历树结构
// 参数u表示当前节点，参数f表示当前节点的父节点
void dfs1(i32 u, i32 f)
{
    // 将当前节点的父节点设置为f
    fa[u] = f;
    // 将当前节点的深度设置为父节点深度加1
    depth[u] = depth[f] + 1;
    // 初始化当前节点的子树大小为1（即包含自身）
    sz[u] = 1;
    // 遍历当前节点的所有邻接节点v
    for (auto v : g[u])
    {
        // 如果邻接节点v是当前节点的父节点，则跳过
        if (v == f)
            continue;
        // 递归调用dfs1函数，处理子节点v，并将当前节点u作为v的父节点
        dfs1(v, u);
        // 将子节点v的子树大小累加到当前节点的子树大小中
        sz[u] += sz[v];
        // 如果子节点v的子树大小大于当前节点重儿子的子树大小，则更新重儿子为v
        if (sz[son[u]] < sz[v])
            son[u] = v;
    }
}

// 定义一个深度优先搜索函数dfs2，用于处理树的重链剖分
void dfs2(i32 u, i32 tp)
{
    // 将当前节点u的链顶设置为tp
    top[u] = tp;
    // 如果当前节点u没有重儿子（即没有子节点），则直接返回
    if (!son[u])
        return;
    // 递归调用dfs2处理当前节点u的重儿子，链顶仍然为tp
    dfs2(son[u], tp);
    // 遍历当前节点u的所有邻接节点v
    for (auto v : g[u])
    {
        // 如果邻接节点v是当前节点u的父节点或者重儿子，则跳过
        if (v == fa[u] || v == son[u])
            continue;
        // 递归调用dfs2处理邻接节点v，链顶设置为v自身
        dfs2(v, v);
    }
}

// 函数 lca 用于计算树中两个节点 u 和 v 的最近公共祖先（LCA）
// 参数 u 和 v 分别表示树中的两个节点
// 返回值是节点 u 和 v 的最近公共祖先的节点编号
i32 lca(i32 u, i32 v)
{
    // 当 u 和 v 不在同一个重链上时，循环继续
    while (top[u] != top[v])
    {
        // 如果 u 所在重链的顶端节点的深度小于 v 所在重链的顶端节点的深度
        if (depth[top[u]] < depth[top[v]])
            // 交换 u 和 v，使得 u 始终是深度较大的节点
            swap(u, v);
        // 将 u 移动到其所在重链的顶端节点的父节点
        u = fa[top[u]];
    }
    // 当 u 和 v 在同一个重链上时，返回深度较小的节点作为最近公共祖先
    return depth[u] < depth[v] ? u : v;
}

signed main()
{
    IOSIO;
    i32 n, m, s;
    cin >> n >> m >> s;
    for (i32 i = 0; i < n - 1; i++)
    {
        i32 u, v;
        cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }

    depth[0] = 0;
    dfs1(s, 0);
    dfs2(s, s);

    for (i32 i = 0; i < m; i++)
    {
        i32 u, v;
        cin >> u >> v;
        cout << lca(u, v) << endl;
    }
    return 0;
}
```
