[[字典树]]（Trie树，又称前缀树）相关内容。

# 原理

- 儿子数组 $ch[p][j]$ 存储从节点 $p$ 沿着j这条边走到的子节点，边为26个小写字母对应的映射值0~25
	- 每个结点最多26个分叉
- 计数数组 $cnt[p]$ 存储以节点 $p$ 结尾的单词的插入数
- 节点编号 $idx$ 用来给节点编号
# 题目

[P8306 【模板】字典树](https://www.luogu.com.cn/problem/P8306)

## CODE

```CPP
// https://www.luogu.com.cn/problem/P8306
#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define IOS ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#define rep(i, j, k) for (int i = (j); i < (k); i++)
#define all(x) x.begin(), x.end()
#define pii pair<int, int>

// - 儿子数组ch[p][j]存储从节点p沿着j这条边走到的子节点，边为26个小写字母对应的映射值0~25
// 	- 每个结点最多26个分叉
// - 计数数组cnt[p]存储以节点p结尾的单词的插入数
// - 节点编号idx用来给节点编号

const int N = 3e6 + 10;
int ch[N][62]; // 前26是小写，后26是大写，最后10是数字
int cnt[N];
int idx = 0;

int get(char ch) {
    if (ch >= 'a' && ch <= 'z') return ch - 'a';
    if (ch >= 'A' && ch <= 'Z') return ch - 'A' + 26;
    return ch - '0' + 52;
}

void insert(const string &s) {
    int p = 0;
    for (int i = 0; i < s.length(); i++) {
        int j = get(s[i]);
        if (!ch[p][j]) ch[p][j] = ++idx;
        p = ch[p][j];
        cnt[p]++;
    }
}

int find(const string &s) {
    int p = 0;
    for (int i = 0; i < s.length(); i++) {
        int j = get(s[i]);
        if (!ch[p][j]) return 0;
        p = ch[p][j];
    }
    return cnt[p];
}

string s;
void solve() {
    int n, m;
    cin >> n >> m;
    rep(i, 0, n) {
        cin >> s;
        insert(s);
    }
    rep(i, 0, m) {
        cin >> s;
        find(s);
        cout << find(s) << endl;
    }
    // cout << idx << endl;
    for (int i = 0; i <= idx; i++) {
        cnt[i] = 0;
        for (int j = 0; j < 62; j++) ch[i][j] = 0;
    }
    idx = 0;
}

int main() {
    IOS;
    fill_n(cnt, N, 0);
    rep(i, 0, N) fill_n(ch[i], 62, 0);
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
```
