# [135. 最大子序和](https://www.acwing.com/problem/content/137/)

利用单调队列维护前缀和

```CPP
#include <bits/stdc++.h>
using namespace std;

using i64 = long long;
const int N = 3e5 + 10;
i64 s[N];
i64 q[N]; // 存储s的下标，并且s[q[i]]是递增的
int main()
{
    int n, m;
    cin >> n >> m;
    s[0] = 0;
    for (int i = 1; i <= n; i++)
    {
        cin >> s[i];
        s[i] += s[i - 1];
    }

    int hh = 0, tt = 0;
    i64 ans = -1e18;
    for (int i = 1; i <= n; i++)
    {
        // 先把多余的删去
        while (i - q[hh] > m)
            hh++;
        // 当前前缀和减去队列维护的对应值，即最小的s，令ans最大
        ans = max(ans, s[i] - s[q[hh]]);
        // 然后更新队列，队尾比当前s大的理论上全都没用，那么都删掉
        while (hh <= tt && s[q[tt]] > s[i])
            tt--;
        q[++tt] = i;
    }
    cout << ans;

    return 0;
}
```

# [289. 环路运输](https://www.acwing.com/problem/content/description/291/)

## 思路

优先队列存储$A_{j}-j$的最值，降序

```CPP
#include <bits/stdc++.h>
using namespace std;

/*
 *@Author: lulaalu
 *@Date: 2024-12-06 16:29:26
 *@Difficulty:中等
 *@Website:https://www.acwing.com/problem/content/description/291/
 */

using i64 = long long;
using i32 = int;

const int N = 2e6 + 10;
i64 s[N];
i32 q[N];
i32 n;
signed main()
{
    cin >> n;
    for (i32 i = 1; i <= n; i++)
    {
        cin >> s[i];
        s[i + n] = s[i];
    }

    i32 hh = 0, tt = -1;
    i64 ans = 0;
    for (i32 i = 1; i <= 2 * n; i++)
    {
        while (hh <= tt && i - q[hh] > n / 2)
            hh++;
        ans = max(ans, i + s[i] + s[q[hh]] - q[hh]);
        while (hh <= tt && s[q[tt]] - q[tt] < s[i] - i)
            tt--;
        q[++tt] = i;
    }
    cout << ans;
    return 0;
}
```
