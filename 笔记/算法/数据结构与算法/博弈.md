# SG函数
```CPP
sg[0] = 0; // 必败态
for (int i = 0; i < n; i++) {
	vector<bool> appear(n, false);
	for (int j = 0; j < i; j++) appear[sg[j]] = true;

	for (int j = 0; j < m; j++) {
		if (!appear[j]) {
			sg[i] = j;
			break;
		}
	}
}
```
