一种分治算法，可用于求区间的数字种类

例题：[Codeforces Round 974 (Div. 3) H. Robin Hood Archery](https://codeforces.com/contest/2014)

题意：求区间内是否有出现奇数次的数

# 算法流程

首先明确这是离线算法，然后，按分治的方式走

1. 分成 $\sqrt{n}$ 份，每份大小为 $\sqrt{n}$  
2. 将要查询的区间排序
	- 先按照 $l$ 排，但是是 $\frac{l}{\sqrt{n}}$ ，小的在前
3. 然后双指针，每次就将指针左右移到对应的区域，维护每个数出现的次数

时间复杂度是 $O(n\sqrt{n})$ 

大致的分析如下：

看上面的排序方式，可以知道在每一块内，$l$ 指针只会在 $\sqrt{n}$ 的范围内反复移动，所有的左指针的移动时间就是 $n\sqrt{n}$ 

然后看右指针，在同一块内的右指针是递增的，那么这一块内右指针移动最高是 $n$ ，下一块的右指针移动最高就是 $n-\sqrt{n}$ ，再下一块就是 $n - 2\sqrt{n}$ 
总时间就是$n\sqrt{n}-\frac{n*(n-1)}{2}*\sqrt{n}$ 

左右一算总时间就是 $O(n\sqrt{n})$ 

# CODE

```CPP
// H. Robin Hood Archery
// Codeforces Round 974 (Div. 3) https://codeforces.com/contest/2014
#pragma GCC optimize(2)
#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define IOS ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#define rep(i, j, k) for (int i = (j); i < (k); i++)
#define all(x) x.begin(), x.end()
#define pii pair<int, int>
#define ull unsigned long long

const int N = 2e5 + 10;
array<int, 3> q[N];// 离线，用个id存答案

int a[N], ans[N];
int cnt[1000001];
int B, odd;
// 排序
bool cmp(array<int, 3> &a, array<int, 3> &b) {
    if (a[0] / B != b[0] / B) return a[0] < b[0];
    return a[1] < b[1];
}

// 扩展一个数
void add(int x) {
    // cout << "add(" << x << ")" << endl;
    odd -= (cnt[x] & 1);
    cnt[x]++;
    odd += (cnt[x] & 1);
}

// 删除一个数
void del(int x) {
    // cout << "del(" << x << ")" << endl;
    odd -= (cnt[x] & 1);
    cnt[x]--;
    odd += (cnt[x] & 1);
}

void solve() {
    odd = 0;
    int n, m;
    cin >> n >> m;
    B = sqrt(n);
    rep(i, 1, n + 1) cin >> a[i];
    rep(i, 1, m + 1) {
        cin >> q[i][0] >> q[i][1];
        q[i][2] = i;
    }
    sort(q + 1, q + m + 1, cmp);

    int ll, rr;
    // 注意++--
    for (int i = 1, l = 1, r = 0; i <= m; i++) {
        while (l > q[i][0]) add(a[--l]);
        while (r < q[i][1]) del(a[++r]);
        while (l < q[i][0]) del(a[l++]);
        while (r > q[i][1]) add(a[r--]);
        ans[q[i][2]] = odd;
        // cout << "l,r : " << l << "," << r << "  odd=" << odd << endl;
        if (i == m) { ll = l, rr = r; }
    }

    for (int i = ll; i <= rr; i++) {
        del(a[i]);
    }

    for (int i = 1; i <= m; i++) {
        if (ans[i])
            cout << "NO" << endl;
        else
            cout << "YES" << endl;
    }
}

int main() {
    IOS;
    fill_n(cnt, 1000001, 0);
    int t;
    cin >> t;
    while (t--) solve();
    return 0;
}
```
