# 链式前向星

需要的结构：
1. $head$ 数组，$head[i]$ 表示 $i$ 点的头边
2. $next$ 数组，$next[i]$ 表示第 $i$ 条边的下一条边的编号
3. $to$ 数组，$to[i]$ 表示第 $i$ 条边去往的点
4. $cnt$ ，表示边的编号

需要的函数：
1. $clear()$ ：清空

```CPP
void clear(int n) {
    cnt = 1;
    fill_n(head, n, 0);
}
```

2. $addEdge(int u,int v)$ ：加边

当前边插入进 $u$ 和 $u$ 头边

作为 $u$ 的头边($head[u] = cnt++$)

并去往 $u$ 的原本的头边($next[cnt] = head[u]$)

当前边指向 $v$ 点($to[cnt] = v$)

```CPP
void add(int u, int v, int w) {
    next[cnt] = head[u];
    to[cnt] = v;
    weight[cnt] = w; // 如果需要的话
    head[u] = cnt++;
}
```

## 链式前向星的遍历

通过 $u$ 获得 $u$ 的头边，用 $next$ 数组遍历与 $u$ 相连的其他边和点

```CPP
void dfs(int u, int fa) {
    cout << u << endl;
    for (int e = head[u]; e > 0; e = next[e]) {// e表示边的编号
        if (to[e] == fa) continue;
        dfs(to[e], u);
    }
}
```

# 拓扑排序

每次删除入度为0的点

[模板题](https://www.nowcoder.com/practice/88f7e156ca7d43a1a535f619cd3f495c)

邻接表建图：
```CPP
void TopologicalSort1() {
    int n, m, u, v;
    cin >> n >> m;
    vector<vi> g(n + 1, vi(0)); // 邻接表
    vi indegree(n + 1, 0);      // 入度表
    rep(i, 0, m) {
        cin >> u >> v;
        g[u].push_back(v);
        indegree[v]++;
    }
    queue<int> q;
    rep(i, 1, n + 1) if (indegree[i] == 0) q.push(i);
    vi ans;
    vector<bool> vis(n + 1, false);
    while (!q.empty()) {
        u = q.front();
        q.pop();
        ans.push_back(u);
        if (vis[u]) break;
        vis[u] = true;
        for (int &v : g[u]) {
            if (--indegree[v] == 0) q.push(v);
        }
    }
    if (ans.size() == n) {
        for (int i = 0; i < n; i++) cout << ans[i] << " "[i == n - 1];
    } else {
        cout << -1;
    }
}
```

```CPP
void TopologicalSort2() {
    int n, m, u, v;
    cin >> n >> m;
    vi head(n + 2, 0); // 链表头
    vi next(m + 2, 0); // 链表下一条边
    vi to(m + 2, 0);   // 链表指向的点
    int cnt = 1;
    vi indegree(n + 1, 0); // 入度表
    auto addEdge = [&](int u, int v) {
        to[cnt] = v;
        next[cnt] = head[u];
        head[u] = cnt++;
    };
    rep(i, 0, m) {
        cin >> u >> v;
        indegree[v]++;
        addEdge(u, v);
    }

    queue<int> q;
    rep(i, 1, n + 1) if (indegree[i] == 0) q.push(i);
    vi ans;
    vector<bool> vis(n + 1, false);
    while (!q.empty()) {
        u = q.front();
        q.pop();
        ans.push_back(u);
        if (vis[u]) break;
        vis[u] = true;
        for (int e = head[u]; e > 0; e = next[e]) {
            if (--indegree[to[e]] == 0) q.push(to[e]);
        }
    }

    if (ans.size() == n) {
        for (int i = 0; i < n; i++) cout << ans[i] << " "[i == n - 1];
    } else {
        cout << -1;
    }
}
```
