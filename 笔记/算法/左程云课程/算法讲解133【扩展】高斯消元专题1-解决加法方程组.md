用于解线性方程组

每次将对应列比较大的数字换到上方，为了避免溢出

# 解的几种情况

1. 唯一解：主元都是1
2. 多解：主元为0的行的最后一列为0
3. 无解：主元为0的行的最后一列不为0

# 某些情况

表达式冗余，就是三个变量四个表达式，补一个x4，最后看前三列的主元，并且最后一列的自由元对应的值为0

// https://www.luogu.com.cn/problem/P3389
// https://www.luogu.com.cn/problem/P2455
#include <bits/stdc++.h>
using namespace std;

#define YES "YES"
#define NO "NO"
#define Yes "Yes"
#define No "No"
#define F first
#define S second
#define int long long
#define ull unsigned long long
#define endl "\n"
#define IOS ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#define rep(i, j, k) for (int i = (j); i < (k); i++)
#define all(x) x.begin(), x.end()
#define vi vector<int>
#define pii pair<int, int>
#define mat vector<vector<double>>

double zero = 1e-7;
mat m;
vector<double> gauss(int n, mat &m) {
    rep(i, 1, n + 1) {
        int mx = i;
        rep(j, 1, n + 1) {
            if (j < i && abs(m[j][j]) >= zero) continue; // 如果是当前行的上面且不是自由元的情况，就跳过
            if (abs(m[mx][i]) < abs(m[j][i])) mx = j;    // 找到第i列的最大值
        }
        if (i != mx) swap(m[i], m[mx]);
        if (abs(m[i][i]) >= zero) { // 保证换上来的位置是大于0的，需要消元
            double tmp = m[i][i];
            rep(j, i, n + 2) m[i][j] /= tmp;

            rep(j, 1, n + 1) { // 把跟自己不一样的行都变成0
                if (i != j) {
                    double rate = m[j][i] / m[i][i];
                    rep(k, i, n + 2) {
                        m[j][k] -= m[i][k] * rate;
                    }
                }
            }
        }
    }

    vector<double> ans = {0}; // 0 表示有解，1表示无解，2表示多解
    rep(i, 1, n + 1) {
        if (abs(m[i][i]) <= zero) {
            if (abs(m[i][n + 1]) > zero) {
                ans[0] = 1;
                return ans;
            } else
                ans[0] = 2;
        }
    }
    if (ans[0] > zero) return ans;
    rep(i, 1, n + 1) ans.push_back(m[i][n + 1]);
    return ans;
}

signed main() {
    IOS;
    int n;
    cin >> n;
    m.resize(n + 1, vector<double>(n + 2));
    rep(i, 1, n + 1) rep(j, 1, n + 2) cin >> m[i][j];
    vector<double> ans = gauss(n, m);
    if (abs(ans[0]) <= zero) {
        rep(i, 1, n + 1) {
            cout << "x" << i << "=";
            cout << setprecision(2) << fixed << ans[i] << endl;
        }
    } else if (abs(ans[0] - 1) <= zero)
        cout << -1;
    else
        cout << 0;
    return 0;
}
