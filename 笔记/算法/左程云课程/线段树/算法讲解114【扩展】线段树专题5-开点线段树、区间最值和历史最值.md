# 开点[[../../../数据结构与算法/线段树/线段树]]

根本思路就是，记录当前节点的左右结点的编号

只需要判断有没有开左右结点，再在一些下传操作时下传 $l[i],r[i]$ 而不是 $i<<1$ 和 $i<<1|1$ 

可以用来无脑处理一些要离散化的信息

如[2276. 统计区间中的整数数目](https://leetcode.cn/problems/count-integers-in-intervals/)

AC代码：

```C++
// https://leetcode.cn/problems/count-integers-in-intervals/description/
const int N = 7e5;
const int n = 1e9;
class CountIntervals {
private:
    int sum[N];
    int l[N], r[N];
    int cnt = 1;

public:
    CountIntervals() {
        fill_n(sum, N, 0);
        fill_n(l, N, 0);
        fill_n(r, N, 0);
        cnt = 1;
    }
    void up(int i) {
        // cout << "up(" << i << ")" << endl;
        sum[i] = 0;
        if (l[i]) sum[i] += sum[l[i]];
        if (r[i]) sum[i] += sum[r[i]];
    }
    void func_add(int jobl, int jobr, int il, int ir, int i) {
        if (sum[i] == ir - il + 1) return;
        if (jobl <= il && ir <= jobr) {
            sum[i] = ir - il + 1;
        } else {
            int mid = il + ir >> 1;
            if (jobl <= mid) {
                if (!l[i]) l[i] = ++cnt;
                func_add(jobl, jobr, il, mid, l[i]);
            }
            if (jobr > mid) {
                if (!r[i]) r[i] = ++cnt;
                func_add(jobl, jobr, mid + 1, ir, r[i]);
            }
            up(i);
        }
    }

    void add(int left, int right) {
        func_add(left, right, 1, n, 1);
        cout << endl;
    }

    int count() {
        return sum[1];
    }
};
```

## 题目3 区间最值操作

线段树的区间最值操作

给定一个长度为n的数组arr，实现支持以下三种操作的结构

操作 0 l r x : 把 $arr[l..r]$ 范围的每个数v，更新成min(v, x)

操作 1 l r   : 查询 $arr[l..r]$ 范围上的最大值

操作 2 l r   : 查询 $arr[l..r]$ 范围上的累加和

### 思路

维护区间最值，最值个数，区间次大值

```CPP
// 吉如一线段树
// 测试链接 : https://acm.hdu.edu.cn/showproblem.php?pid=5306

#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define IOS ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#define rep(i, j, k) for (int i = (j); i < (k); i++)
#define all(x) x.begin(), x.end()
#define pii pair<int, int>

#define int long long
const int N = 1e6 + 10;
const int MI = -1e10;
int arr[N];
int sum[N << 2], mx[N << 2], sem[N << 2], cnt[N << 2];
int n, m;

void up(int i) {
    int l = i << 1;
    int r = i << 1 | 1;
    sum[i] = sum[l] + sum[r];
    mx[i] = max(mx[l], mx[r]);
    if (mx[l] > mx[r]) {
        cnt[i] = cnt[l];
        sem[i] = max(mx[r], sem[l]);
    } else if (mx[l] < mx[r]) {
        cnt[i] = cnt[r];
        sem[i] = max(mx[l], sem[r]);
    } else {
        cnt[i] = cnt[l] + cnt[r];
        sem[i] = max(sem[l], sem[r]);
    }
}

// 一定是最大值被压成v，并且v严格大于次大值
void lazy(int i, int v) {
    if (v < mx[i]) {
        sum[i] -= (mx[i] - v) * cnt[i];
        mx[i] = v;
    }
}

void down(int i) {
    lazy(i << 1, mx[i]);
    lazy(i << 1 | 1, mx[i]);
}

void setMin(int jobl, int jobr, int jobv, int l, int r, int i) {
    if (jobv >= mx[i]) {
        return;
    }
    if (jobl <= l && r <= jobr && sem[i] < jobv) {
        lazy(i, jobv);
    } else {
        // 1. 任务没全包  2. jobv<=sem[i]
        down(i);

        int mid = l + r >> 1;
        if (jobl <= mid) setMin(jobl, jobr, jobv, l, mid, i << 1);
        if (jobr > mid) setMin(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        up(i);
    }
}

int queryMax(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        return mx[i];
    }
    down(i);
    int mid = l + r >> 1;
    int ans = MI;
    if (jobl <= mid) ans = max(ans, queryMax(jobl, jobr, l, mid, i << 1));
    if (jobr > mid) ans = max(ans, queryMax(jobl, jobr, mid + 1, r, i << 1 | 1));
    return ans;
}

int querySum(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        return sum[i];
    }
    down(i);

    int mid = (l + r) >> 1;
    int ans = 0;
    if (jobl <= mid) ans += querySum(jobl, jobr, l, mid, i << 1);
    if (jobr > mid) ans += querySum(jobl, jobr, mid + 1, r, i << 1 | 1);
    return ans;
}

void build(int l, int r, int i) {
    if (l == r) {
        sum[i] = arr[l];
        mx[i] = arr[l];
        cnt[i] = 1;
        sem[i] = MI;
        return;
    }
    int mid = l + r >> 1;
    build(l, mid, i << 1);
    build(mid + 1, r, i << 1 | 1);
    up(i);
}
void solve() {
    cin >> n;
    cin >> m;
    rep(i, 1, n + 1) cin >> arr[i];
    build(1, n, 1);
    rep(i, 0, m) {
        int op, l, r, x;
        cin >> op;
        // cout << "case" << endl;
        switch (op) {
        case 0:
            cin >> l >> r >> x;
            setMin(l, r, x, 1, n, 1);
            break;
        case 1:
            cin >> l >> r;
            cout << queryMax(l, r, 1, n, 1) << endl;
            break;
        case 2:
            cin >> l >> r;
            cout << querySum(l, r, 1, n, 1) << endl;
            break;
        }
    }
}

signed main() {
    IOS;
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
```
## 题目4 范围增加操作 + 区间最值操作

给定一个长度为n的数组arr，实现支持以下四种操作的结构

操作 0 l r x : 把arr[l..r]范围的每个数v，增加x

操作 1 l r x : 把arr[l..r]范围的每个数v，更新成min(v, x)

操作 2 l r   : 查询arr[l..r]范围上的最大值

操作 3 l r   : 查询arr[l..r]范围上的累加和

### 思路

正常分开处理就行，lazy_add和lazy_min注意在down中的调用顺序

### code1
```CPP
// 吉如一线段树
#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define IOS ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#define rep(i, j, k) for (int i = (j); i < (k); i++)
#define all(x) x.begin(), x.end()
#define pii pair<int, int>

const int N = 2e5;
const int MI = -1e9;
int arr[N];
int sum[N << 2];
int mx[N << 2];
int cnt[N << 2];
int sem[N << 2];
int add[N << 2];

void up(int i) {
    int l = i << 1, r = i << 1 | 1;
    sum[i] = sum[l] + sum[r];
    mx[i] = max(mx[l], mx[r]);
    if (mx[l] > mx[r]) {
        cnt[i] = cnt[l];
        sem[i] = max(sem[l], mx[r]);
    } else if (mx[l] < mx[r]) {
        cnt[i] = cnt[r];
        sem[i] = max(sem[r], mx[l]);
    } else {
        cnt[i] = cnt[l] + cnt[r];
        sem[i] = max(sem[l], sem[r]);
    }
}

void lazy_add(int v, int l, int r, int i) {
    add[i] += v;
    sum[i] += v * (r - l + 1);
    mx[i] += v;
    if (sem[i] != MI) sem[i] += v;
}

void lazy_min(int i, int v) {
    if (v < mx[i]) {
        sum[i] -= (mx[i] - v) * cnt[i];
        mx[i] = v;
    }
}

void down(int l, int r, int i) {
    /*
    注意次序问题,lazymin的实现是v在mx和sem中间时能正确的更新、
    但是在add的更新中，如果先更新add再更新min，那么min的更新会覆盖掉add的更新
    */
    if (add[i] != 0) {
        int mid = l + r >> 1;
        lazy_add(add[i], l, mid, i << 1);
        lazy_add(add[i], mid + 1, r, i << 1 | 1);
        add[i] = 0;
    }
    lazy_min(i << 1, mx[i]);
    lazy_min(i << 1 | 1, mx[i]);
}

void func_setmin(int jobl, int jobr, int jobv, int l, int r, int i) {
    if (jobv >= mx[i]) return;
    if (jobl <= l && r <= jobr && jobv > sem[i]) {
        lazy_min(i, jobv);
    } else {
        down(l, r, i);
        int mid = (l + r) >> 1;
        if (jobl <= mid) func_setmin(jobl, jobr, jobv, l, mid, i << 1);
        if (jobr > mid) func_setmin(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        up(i);
    }
}

void func_add(int jobl, int jobr, int jobv, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        lazy_add(jobv, l, r, i);
    } else {
        down(l, r, i);
        int mid = (l + r) >> 1;
        if (jobl <= mid) func_add(jobl, jobr, jobv, l, mid, i << 1);
        if (jobr > mid) func_add(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        up(i);
    }
}

int func_querymax(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        return mx[i];
    } else {
        down(l, r, i);
        int mid = (l + r) >> 1;
        int ans = MI;
        if (jobl <= mid) ans = max(ans, func_querymax(jobl, jobr, l, mid, i << 1));
        if (jobr > mid) ans = max(ans, func_querymax(jobl, jobr, mid + 1, r, i << 1 | 1));
        return ans;
    }
}

int func_querysum(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        return sum[i];
    } else {
        down(l, r, i);
        int mid = (l + r) >> 1;
        int ans = 0;
        if (jobl <= mid) ans += func_querysum(jobl, jobr, l, mid, i << 1);
        if (jobr > mid) ans += func_querysum(jobl, jobr, mid + 1, r, i << 1 | 1);
        return ans;
    }
}

void build(int l, int r, int i) {
    add[i] = 0;
    if (l == r) {
        sum[i] = mx[i] = arr[l];
        sem[i] = MI;
        cnt[i] = 1;
        return;
    }
    int mid = (l + r) >> 1;
    build(l, mid, i << 1);
    build(mid + 1, r, i << 1 | 1);
    up(i);
}

void solve() {
    int n, m;
    cin >> n;
    rep(i, 1, n + 1) cin >> arr[i];
    cin >> m;
    build(1, n, 1);
    rep(i, 0, m) {
        int op, l, r, x;
        cin >> op;
        switch (op) {
        case 0:
            cin >> l >> r >> x;
            func_add(l, r, x, 1, n, 1);
            break;
        case 1: {
            cin >> l >> r >> x;
            func_setmin(l, r, x, 1, n, 1);
            break;
        }
        case 2: {
            cin >> l >> r;
            cout << func_querymax(l, r, 1, n, 1) << endl;
            break;
        }
        case 3: {
            cin >> l >> r;
            cout << func_querysum(l, r, 1, n, 1) << endl;
            break;
        }
        }
    }
}
int main() {
    IOS;
    solve();
    return 0;
}
```

### code2

将addlazy和minlazy合成一个，用一个othadd数组表示非最大值的懒标记

在func_setmin中的lzay用于处理mx和sem之间的剪枝，那么就是mx懒住

在func_add中lazy的增加是所有值的增加，那么mxadd和othadd同时懒住

注意down()中，由于父亲下发懒标记时，父亲很可能已经被修改最大值

但是父亲的最大值修改了，子节点的最大值也要修改，所以取左右结点中mx大的下发mxadd和othadd，小的下发othadd和mxadd

```CPP
#include <bits/stdc++.h>
using namespace std;

#define endl "\n"
#define IOS ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#define rep(i, j, k) for (int i = (j); i < (k); i++)
#define all(x) x.begin(), x.end()
#define pii pair<int, int>

const int N = 2e5;
const int MI = -1e9;
int arr[N];
int sum[N << 2];
int mx[N << 2];
int cnt[N << 2];
int sem[N << 2];
int mxadd[N << 2];
int othadd[N << 2];

void up(int i) {
    int l = i << 1, r = i << 1 | 1;
    sum[i] = sum[l] + sum[r];
    mx[i] = max(mx[l], mx[r]);
    if (mx[l] > mx[r]) {
        cnt[i] = cnt[l];
        sem[i] = max(sem[l], mx[r]);
    } else if (mx[l] < mx[r]) {
        cnt[i] = cnt[r];
        sem[i] = max(sem[r], mx[l]);
    } else {
        cnt[i] = cnt[l] + cnt[r];
        sem[i] = max(sem[l], sem[r]);
    }
}

void lazy(int l, int r, int i, int mxadd_, int othadd_) {
    int len = r - l + 1;
    sum[i] += mxadd_ * cnt[i] + othadd_ * (len - cnt[i]);
    mx[i] += mxadd_;
    if (sem[i] != MI) sem[i] += othadd_;
    mxadd[i] += mxadd_;
    othadd[i] += othadd_;
}
/*
父亲下发懒更新任务
*/
void down(int l, int r, int i) {
    int li = i << 1, ri = i << 1 | 1;
    int mid = l + r >> 1;
    int tmp = max(mx[li], mx[ri]);
    // 父亲可能已经改变
    if (tmp == mx[li]) { // 如果左边是最大值，下传时需要更改左边的最大值
        lazy(l, mid, li, mxadd[i], othadd[i]);
    } else { // 否则就都用其他值的最大值来修改
        lazy(l, mid, li, othadd[i], othadd[i]);
    }
    if (tmp == mx[ri]) {
        lazy(mid + 1, r, ri, mxadd[i], othadd[i]);
    } else {
        lazy(mid + 1, r, ri, othadd[i], othadd[i]);
    }
    mxadd[i] = othadd[i] = 0;
}

void func_setmin(int jobl, int jobr, int jobv, int l, int r, int i) {
    if (jobv >= mx[i]) return;
    if (jobl <= l && r <= jobr && jobv > sem[i]) {
        lazy(l, r, i, jobv - mx[i], 0);
    } else {
        down(l, r, i);
        int mid = (l + r) >> 1;
        if (jobl <= mid) func_setmin(jobl, jobr, jobv, l, mid, i << 1);
        if (jobr > mid) func_setmin(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        up(i);
    }
}

void func_add(int jobl, int jobr, int jobv, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        lazy(l, r, i, jobv, jobv);
    } else {
        down(l, r, i);
        int mid = (l + r) >> 1;
        if (jobl <= mid) func_add(jobl, jobr, jobv, l, mid, i << 1);
        if (jobr > mid) func_add(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        up(i);
    }
}

int func_querymax(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        return mx[i];
    } else {
        down(l, r, i);
        int mid = (l + r) >> 1;
        int ans = MI;
        if (jobl <= mid) ans = max(ans, func_querymax(jobl, jobr, l, mid, i << 1));
        if (jobr > mid) ans = max(ans, func_querymax(jobl, jobr, mid + 1, r, i << 1 | 1));
        return ans;
    }
}

int func_querysum(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        return sum[i];
    } else {
        down(l, r, i);
        int mid = (l + r) >> 1;
        int ans = 0;
        if (jobl <= mid) ans += func_querysum(jobl, jobr, l, mid, i << 1);
        if (jobr > mid) ans += func_querysum(jobl, jobr, mid + 1, r, i << 1 | 1);
        return ans;
    }
}

void build(int l, int r, int i) {
    mxadd[i] = 0;
    othadd[i] = 0;
    if (l == r) {
        sum[i] = mx[i] = arr[l];
        sem[i] = MI;
        cnt[i] = 1;
        return;
    }
    int mid = (l + r) >> 1;
    build(l, mid, i << 1);
    build(mid + 1, r, i << 1 | 1);
    up(i);
}

void solve() {
    int n, m;
    cin >> n;
    rep(i, 1, n + 1) cin >> arr[i];
    cin >> m;
    build(1, n, 1);
    rep(i, 0, m) {
        int op, l, r, x;
        cin >> op;
        switch (op) {
        case 0:
            cin >> l >> r >> x;
            func_add(l, r, x, 1, n, 1);
            break;
        case 1: {
            cin >> l >> r >> x;
            func_setmin(l, r, x, 1, n, 1);
            break;
        }
        case 2: {
            cin >> l >> r;
            cout << func_querymax(l, r, 1, n, 1) << endl;
            break;
        }
        case 3: {
            cin >> l >> r;
            cout << func_querysum(l, r, 1, n, 1) << endl;
            break;
        }
        }
    }
}

int main() {
    IOS;
    solve();
    return 0;
}
```

## 题目5 3+4+历史最值

给定两个长度都为n的数组A和B，一开始两个数组完全一样

任何操作做完，都更新B数组，B[i] = max(B[i],A[i])

实现以下五种操作，一共会调用m次

操作 1 l r v : A[l..r]范围上每个数加上v

操作 2 l r v : A[l..r]范围上每个数A[i]变成min(A[i],v)

操作 3 l r   : 返回A[l..r]范围上的累加和

操作 4 l r   : 返回A[l..r]范围上的最大值

操作 5 l r   : 返回B[l..r]范围上的最大值

1 <= n、m <= 5 * 10^5

测试链接 : https://www.luogu.com.cn/problem/P6242

## 思路
比题目4多维护几个信息

历史最大值
历史最大值的最大增幅
历史其他值的最大增幅

由于是范围上的最大值，所以不能通过只维护一个历史最大值来获取答案
这样会造成 更新被懒住，下层历史最值不能及时更新
