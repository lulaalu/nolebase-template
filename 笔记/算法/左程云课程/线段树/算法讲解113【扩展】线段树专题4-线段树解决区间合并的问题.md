例题：

## [SCOI2010] 序列操作

### 题目描述

lxhgww 最近收到了一个 $01$ 序列，序列里面包含了 $n$ 个数，下标从 $0$ 开始。这些数要么是 $0$，要么是 $1$，现在对于这个序列有五种变换操作和询问操作：

- `0 l r` 把 $[l, r]$ 区间内的所有数全变成 $0$；
- `1 l r` 把 $[l, r]$ 区间内的所有数全变成 $1$；
- `2 l r` 把 $[l,r]$ 区间内的所有数全部取反，也就是说把所有的 $0$ 变成 $1$，把所有的 $1$ 变成 $0$；
- `3 l r` 询问 $[l, r]$ 区间内总共有多少个 $1$；
- `4 l r` 询问 $[l, r]$ 区间内最多有多少个连续的 $1$。

维护

区间前缀0/1的数量
区间中最长0/1的数量
区间后缀0/1的数量

区间1的总数量

### 代码

```CPP
#include <bits/stdc++.h>
using namespace std;

#define int long long
#define ull unsigned long long
#define endl "\n"
#define IOS ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
#define rep(i, j, k) for (int i = (j); i < (k); i++)
#define all(x) x.begin(), x.end()
#define vi vector<int>
typedef pair<int, int> pii;

const int MOD = 1000000007;
const int N = 1e5 + 10;

int a[N];
int pre1[N << 2], pre0[N << 2],
    len1[N << 2], len0[N << 2],
    suf1[N << 2], suf0[N << 2],
    sum1[N << 2];

int change[N << 2];  // 变为0或1
bool update[N << 2]; // 是否有区间修改任务
bool rev[N << 2];    // 取反

void up(int l, int r, int i) {
    int mid = l + r >> 1;
    int ln = mid - l + 1;
    int rn = r - mid;
    int li = i << 1;
    int ri = i << 1 | 1;
    pre1[i] = pre1[li] == ln ? pre1[li] + pre1[ri] : pre1[li]; //*
    pre0[i] = pre0[li] == ln ? pre0[li] + pre0[ri] : pre0[li];
    suf1[i] = suf1[ri] == rn ? suf1[ri] + suf1[li] : suf1[ri];
    suf0[i] = suf0[ri] == rn ? suf0[ri] + suf0[li] : suf0[ri];
    len1[i] = max({len1[li], len1[ri], suf1[li] + pre1[ri]});
    len0[i] = max({len0[li], len0[ri], suf0[li] + pre0[ri]});
    sum1[i] = sum1[li] + sum1[ri];
}

void build(int l, int r, int i) {

    change[i] = 0;
    update[i] = false;
    rev[i] = false;
    if (l == r) {
        pre1[i] = suf1[i] = len1[i] = a[l];
        pre0[i] = suf0[i] = len0[i] = a[l] ^ 1;
        sum1[i] = a[l];
        return;
    }
    int mid = l + r >> 1;
    build(l, mid, i << 1);
    build(mid + 1, r, i << 1 | 1);
    up(l, r, i);
}

void func_lazy_update(int l, int r, int i, int jobv) {
    int len = r - l + 1;
    sum1[i] = len * jobv;
    len1[i] = pre1[i] = suf1[i] = len * jobv;
    len0[i] = pre0[i] = suf0[i] = len * (1 ^ jobv);
    change[i] = jobv;
    update[i] = true;
    rev[i] = false; //*
}

void func_lazy_rev(int l, int r, int i) {
    rev[i] = !rev[i];
    swap(pre1[i], pre0[i]);
    swap(suf1[i], suf0[i]);
    swap(len1[i], len0[i]);
    sum1[i] = r - l + 1 - sum1[i];
}

void down(int l, int r, int i) {
    int mid = l + r >> 1;
    if (update[i]) { // 先处理update的更新
        func_lazy_update(l, mid, i << 1, change[i]);
        func_lazy_update(mid + 1, r, i << 1 | 1, change[i]);
        update[i] = 0;
    }
    if (rev[i]) {
        func_lazy_rev(l, mid, i << 1);
        func_lazy_rev(mid + 1, r, i << 1 | 1);
        rev[i] = 0;
    }
}

void func_update(int jobl, int jobr, int jobv, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        func_lazy_update(l, r, i, jobv);
    } else {
        down(l, r, i);
        int mid = l + r >> 1;
        if (jobl <= mid) func_update(jobl, jobr, jobv, l, mid, i << 1);
        if (mid < jobr) func_update(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        up(l, r, i);
    }
}

void func_rev(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        func_lazy_rev(l, r, i);
    } else {
        down(l, r, i);
        int mid = l + r >> 1;
        if (jobl <= mid) func_rev(jobl, jobr, l, mid, i << 1);
        if (mid < jobr) func_rev(jobl, jobr, mid + 1, r, i << 1 | 1);
        up(l, r, i);
    }
}

int func_query_sum1(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) return sum1[i];

    down(l, r, i);
    int mid = l + r >> 1;
    int ans = 0;
    if (jobl <= mid) ans += func_query_sum1(jobl, jobr, l, mid, i << 1);
    if (mid < jobr) ans += func_query_sum1(jobl, jobr, mid + 1, r, i << 1 | 1);
    return ans;
}

// 返回信息：<len,pre,suf>
tuple<int, int, int> func_query_len1(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) return {len1[i], pre1[i], suf1[i]};

    down(l, r, i);
    int mid = l + r >> 1;
    // 注意符号
    if (jobr <= mid) return func_query_len1(jobl, jobr, l, mid, i << 1);
    if (mid < jobl) return func_query_len1(jobl, jobr, mid + 1, r, i << 1 | 1);

    auto [len1_l, pre1_l, suf1_l] = func_query_len1(jobl, jobr, l, mid, i << 1);
    auto [len1_r, pre1_r, suf1_r] = func_query_len1(jobl, jobr, mid + 1, r, i << 1 | 1);

    int len = max({len1_l, len1_r, suf1_l + pre1_r});

    int ln = mid - max(l, jobl) + 1, rn = min(r, jobr) - mid;
    int pre = pre1_l == ln ? pre1_l + pre1_r : pre1_l;
    int suf = suf1_r == rn ? suf1_l + suf1_r : suf1_r;
    return {len, pre, suf};
}

int n, m, op, l, r;
signed main() {
    IOS;
    cin >> n >> m;
    rep(i, 1, n + 1) cin >> a[i];
    build(1, n, 1);
    rep(i, 0, m) {

        cin >> op >> l >> r;
        l++, r++;
        switch (op) {
        case 0: func_update(l, r, 0, 1, n, 1); break;
        case 1: func_update(l, r, 1, 1, n, 1); break;
        case 2: func_rev(l, r, 1, n, 1); break;
        case 3: cout << func_query_sum1(l, r, 1, n, 1) << endl; break;
        case 4: cout << get<0>(func_query_len1(l, r, 1, n, 1)) << endl; break;
        }
    }

    return 0;
}
```
